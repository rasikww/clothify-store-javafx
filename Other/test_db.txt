CREATE TABLE user (
    user_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50),
    phone_number VARCHAR(20),
    email VARCHAR(255) NOT NULL,
    password_hash VARCHAR(32) NOT NULL,
    is_admin TINYINT(1) NOT NULL DEFAULT 0
);
